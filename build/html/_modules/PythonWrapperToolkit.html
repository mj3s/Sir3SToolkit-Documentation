        """
        Start a transaction with the given session name.

        :param SessionName: A meaningful name to start a transaction. An empty string or None will lead to an error.
        :type SessionName: str
        :return: None
        :rtype: None
        :raises ValueError: If the transaction could not be started.
        
        This method is a wrapper for StartTransaction() from the toolkit.
        """
        isTransactionStarted, message = self.toolkit.StartTransaction(SessionName)
        if not isTransactionStarted:
            print(message)
        else:
            if message == "":
                print("Now you can make changes to the model")
            else:
                print(message)

    # Description : This method is a wrapper method for StartTransaction() from toolkit
    # ============================================================================================================================       
    def StartTransaction(self, SessionName: str):
        isTransactionStarted, message = self.toolkit.StartTransaction(SessionName)
        if not isTransactionStarted:
            print(message)
        else:

            else:
